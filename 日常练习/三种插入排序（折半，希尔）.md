#三种插入排序
##直接插入排序
	#include <bits/stdc++.h>
	#define FOR(i, a, b) for (int i = (a); i <= (b); i++)
	#define FORD(i, a, b) for (int i = (a); i >= (b); i--)
	#define MAX 1050
	#define INF 0x3f3f3f3f
	using namespace std;
	
	void BasicInsertSort(int a[], int n){
	    int tmp[MAX];
	    //复制数组，并清空原数组
	    FOR(i, 0, n)    tmp[i] = a[i], a[i] = -INF;
	    //插入排序，并重建原数组
	    FOR(i, 1, n){
	        int key = tmp[i];
	        FORD(j, i, 1){
	            if(key > a[j - 1]){
	                a[j] = key;
	                break;
	            }
	            else{
	                a[j] = a[j - 1];
	            }
	        }
	    }
	}
	
	int main(){
	    ios::sync_with_stdio(0);
	    cin.tie(0);
	    int n, a[MAX];
	    while(cin >> n){
	        FOR(i, 1, n)  cin >> a[i];
	        BasicInsertSort(a, n);
	        FOR(i, 1, n)
	            i == n? cout << a[i] << endl: cout << a[i] << " ";
	    }
	    return 0;
	}
