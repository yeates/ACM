# 素数筛法
	#include <bits/stdc++.h>
	#define MAX 10001
	#define INF 0x3f3f3f3f
	#define LL long long
	#define Min(a, b) (a < b)? a: b
	#define Max(a, b) (a < b)? b: a
	using namespace std;
	
	int prime[MAX];
	
	int main(){
	    ios::sync_with_stdio(0);
	    cin.tie(0);
	    //筛法(简易版)
	    memset(prime, 0, sizeof(prime));
	    prime[0] = prime[1] = 1;
	    for(int i = 2; i < MAX; i++){
	        if(prime[i] == 1)   continue;
	        int j = i * i;      //可以简单证明不用j = i + i
	        while(j < MAX){
	            prime[j] = 1;
	            j += i;
	        }
	    }
	    int n;
	    while(cin >> n){
	        if(prime[n])    cout << "不是素数" << endl;
	        else cout << "是素数" << endl;
	    }
	    return 0;
	}
